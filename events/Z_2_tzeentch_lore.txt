namespace = tzeentch_lore


character_event = {
	id = tzeentch_lore.101#Drain Magic Power 1"
	desc = tzeentch_lore.101_desc#"Your spell burns through the air, taking from all enemies the ability to cast. "
	is_triggered_only = yes
	picture = GFX_evt_mage_lore_necromancy


	option = {
		name = tzeentch_lore.101a#No
		character_event = { id = z_spell_menu.99996}

		event_target:magic_caster = {
			set_variable = { which = zdmg value = 1 }
			set_variable = { which = zmoraledmg value = 1 }
			change_variable = { which = zdmg which = caster_level }
			change_variable = { which = zmoraledmg which = caster_level }

		}
		random_list = {
			1 = {
				event_target:magic_caster = {
					change_variable = { which = zdmg value = 1 }
					change_variable = { which = zmoraledmg value = 1 }
				}
			}
			1 = {
				event_target:magic_caster = {
					change_variable = { which = zdmg value = 2 }
					change_variable = { which = zmoraledmg value = 2 }
				}
			}
			1 = {
				event_target:magic_caster = {
					change_variable = { which = zdmg value = 3 }
					change_variable = { which = zmoraledmg value = 3 }
				}
			}
			1 = {
				event_target:magic_caster = {
					change_variable = { which = zdmg value = 4 }
					change_variable = { which = zmoraledmg value = 4 }
				}
			}
			1 = {
				event_target:magic_caster = {
					change_variable = { which = zdmg value = 5 }
					change_variable = { which = zmoraledmg value = 5 }
				}
			}
			1 = {
				event_target:magic_caster = {
					change_variable = { which = zdmg value = 6 }
					change_variable = { which = zmoraledmg value = 6 }
				}
			}
			1 = {
				event_target:magic_caster = {
					change_variable = { which = zdmg value = 7 }
					change_variable = { which = zmoraledmg value = 7 }
				}
			}
			1 = {
				event_target:magic_caster = {
					change_variable = { which = zdmg value = 8 }
					change_variable = { which = zmoraledmg value = 8 }
				}
			}
			1 = {
				event_target:magic_caster = {
					change_variable = { which = zdmg value = 9 }
					change_variable = { which = zmoraledmg value = 9 }
				}
			}
		}
		any_character = {
			limit = {
				war_with = root
				any_army = {
					location = { province = event_target:magic_loc }
				}
			}
			random_army = {
				limit = {
					location = { province = event_target:magic_loc }
				}
				leader = {
					magic_points_setup = yes
					save_event_target_as = magic_target
					magic_resist_test = yes
				}
				if = {
					limit = {
						event_target:magic_caster = {
							has_character_flag = resisted
						}
					}
					event_target:magic_caster = {
						clr_character_flag = resisted
						divide_variable = { which = zdmg value = 2 }
						divide_variable = { which = zmoraledmg value = 2 }
					}
				}
				magic_dmg_effect = yes
				morale_dmg_effect = yes
			}
		}
		if = {
			limit = {
				event_target:magic_caster = {
					at_location = event_target:magic_loc
				}
			}
			siege = {
				enemy = {
					magic_dmg_effect = yes
					morale_dmg_effect = yes
				}
			}
		}
	}
}
