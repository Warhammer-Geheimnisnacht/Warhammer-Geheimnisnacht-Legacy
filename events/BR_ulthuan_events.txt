namespace = ulthuan

#Provinces in Ulthuan slowly depopulate even during peace, but much faster during times of war when the province is occupied - first we assign each province in Ulthuan a modifier
province_event = {
	id = ulthuan.0
		title = "Ulthuan"
		desc = "Ulthuan is a nation in decline"
		picture = GFX_evt_council
		hide_window = yes
		is_triggered_only = yes

		trigger = {
			NOT = { has_game_rule = { name = warhammer_ulthuan_depopulation value = off } }
			culture_group = asur_group
				NOT = { has_province_modifier = ulthuan }
			NOT = {
				province_id = 1492 ##Not Lothern, it welcomes visitors from around the world and hence non elves replace elven population losses
			}
			region = world_ulthuan
		}

	option = {
		name = "OK"
			ROOT = {
				add_province_modifier = { name = ulthuan duration = -1 }

			}
	}

}

#Now count tier and above rulers get the event informing them that their province has been depopulated to lightly depopulated##
character_event = {
	id = ulthuan.1
	# Whether through war or just the slow passage of time, the population in this province has fallen. Such is the fate of all elves in Ulthuan, to slowly diminish, but they will never yield to their enemies even so.
	desc = {
		text = ulthuan1desc
		trigger = {
			event_target:elven_province = {
				NOR = {
					has_province_modifier = ulthuan_lightly_depopulated
					has_province_modifier = ulthuan_moderately_depopulated
					has_province_modifier = ulthuan_heavily_depopulated
					has_province_modifier = ulthuan_severely_depopulated
				}
			}
		}
	}
	#Whether through war or just the slow passage of time, the population in this province is now moderately depopulated. Empty villages and mansions dot the country side, their former inhabitants now only a distant memory.#
	desc = {
		text = ulthuan2desc
		trigger = {
			event_target:elven_province = {
				has_province_modifier = ulthuan_lightly_depopulated
			}
		}
	}
	 # Whether through war or just the slow passage of time, the population in this province is now heavily depopulated. The Lords there struggle to maintain a semblance of civilization, and many areas are now devoid of elves entirely, given over to nature and the wild beasts.#
	desc = {
		text = ulthuan3desc
		trigger = {
			event_target:elven_province = {
				has_province_modifier = ulthuan_moderatly_depopulated
			}
		}
	}
	# Whether through war or just the slow passage of time, the population in this province is now severely depopulated. The major noble families still maintain their mansions and the immediate surrounds, but they have lost the ability to control much more than that.#
	desc = {
		text = ulthuan4desc
		trigger = {
			event_target:elven_province = {
				has_province_modifier = ulthuan_heavily_depopulated
			}
		}
	}

	picture = "GFX_evt_elf_city"

	culture_group = asur_group
	only_rulers = yes

	# Select province to depopulate
	immediate = {
		random_demesne_province = {
			limit = {
				has_province_modifier = ulthuan
				NOT = { has_province_modifier = ulthuan_severely_depopulated }
			}
			save_event_target_as = elven_province
		}
	}

	trigger = {
		NOT = { has_game_rule = { name = warhammer_ulthuan_depopulation value = off } }
		higher_tier_than = BARON
		any_demesne_province = {
			AND = {
				has_province_modifier = ulthuan
				NOT = { has_province_modifier = ulthuan_severely_depopulated }
			}
		}
	}

	mean_time_to_happen = {
		months = 2400 #200 years

		modifier = {
			factor = 0.5
			liege = {
				war = yes
				has_raised_levies = ROOT
			}
		}

		modifier = {
			factor = 3
			has_game_rule = { name = warhammer_ulthuan_depopulation value = decreased }
		}
	}

	option = {
		name = OK
		event_target:elven_province = {
			trigger_switch = {
				on_trigger = has_province_modifier

				# heavily -> severly
				ulthuan_heavily_depopulated = {
					add_province_modifier = {
						name = ulthuan_severely_depopulated duration = -1
					}
					remove_province_modifier = ulthuan_heavily_depopulated
				}
				# moderatly -> heavily
				ulthuan_moderately_depopulated = {
					add_province_modifier = {
						name = ulthuan_heavily_depopulated duration = -1
					}
					remove_province_modifier = ulthuan_moderately_depopulated
				}
				# lightly -> moderatly
				ulthuan_lightly_depopulated = {
					add_province_modifier = {
						name = ulthuan_moderately_depopulated duration = -1
					}
					remove_province_modifier = ulthuan_lightly_depopulated
				}
			}
			# none -> lighlty
			if = {
				limit = {
					NOR = {
						has_province_modifier = ulthuan_lightly_depopulated
						has_province_modifier = ulthuan_moderately_depopulated
						has_province_modifier = ulthuan_heavily_depopulated
						has_province_modifier = ulthuan_severely_depopulated
					}
				}
				add_province_modifier = {
					name = ulthuan_lightly_depopulated duration = -1
				}
			}
		}
	}
}

##Now for when a province is occupied, the population decline is even faster##
character_event = {
	id = ulthuan.5
		desc = ulthuan5desc # This province is occupied by enemies and many precious elven lives have been lost.
		picture = "GFX_evt_elf_city"

		only_rulers = yes

		trigger = {
			NOT = { has_game_rule = { name = warhammer_ulthuan_depopulation value = off } }
			culture_group = asur_group
			higher_tier_than = BARON
			any_demesne_title = {
				location = {
					has_province_modifier = ulthuan
				}
				tier = BARON
				NOT = { controlled_by = ROOT }
			}
		}

	mean_time_to_happen = {
		months = 60 #5 years#

		modifier = {
			factor = 5
			has_game_rule = { name = warhammer_ulthuan_depopulation value = decreased }
		}
	}

	option = {
		name = OK
			random_demesne_province={
				if = {
					limit = {
						NOT = { controlled_by = ROOT }
					}
					if = {
						limit = {
							AND = {
								has_province_modifier = ulthuan
								has_province_modifier = ulthuan_heavily_depopulated
							}
						}
						add_province_modifier = { name = ulthuan_severely_depopulated duration = -1 }
						remove_province_modifier = ulthuan_heavily_depopulated
					}
					if = {
						limit = {
							AND = {
								has_province_modifier = ulthuan
								has_province_modifier = ulthuan_moderately_depopulated
							}
						}
						add_province_modifier = { name = ulthuan_heavily_depopulated duration = -1 }
						remove_province_modifier = ulthuan_moderately_depopulated
					}
					if = {
						limit = {
							AND = {
								has_province_modifier = ulthuan
								has_province_modifier = ulthuan_lightly_depopulated
							}
						}
						add_province_modifier = { name = ulthuan_moderately_depopulated duration = -1 }
						remove_province_modifier = ulthuan_lightly_depopulated
					}
					if = {
						limit = {
							AND = {
								has_province_modifier = ulthuan
								NOT = { has_province_modifier = ulthuan_lightly_depopulated }
								NOT = { has_province_modifier = ulthuan_moderately_depopulated }
								NOT = { has_province_modifier = ulthuan_heavily_depopulated }
								NOT = { has_province_modifier = ulthuan_severely_depopulated }
							}
						}
						add_province_modifier = { name = ulthuan_lightly_depopulated duration = -1 }
					}
				}
			}
	}

}

		desc = ulthuan6desc #"Non elves are not affected by the vortex, nor - we suppose - dark elves who are... more aligned with chaos or just have more sex?"
		picture = GFX_evt_council
		hide_window = yes

		trigger = {
			NOT = { has_game_rule = { name = warhammer_ulthuan_depopulation value = off } }
			NOT = { culture_group = asur_group }
			has_province_modifier = ulthuan
				NOT = {
					province_id = 1492 ##Not Lothern, it welcomes visitors from around the world and hence non elves replace elven population losses
				}
			region = world_ulthuan
		}

	mean_time_to_happen = {
		days = 1
	}

	option = {
		name = "OK"
		ROOT = {
				remove_province_modifier = ulthuan
				remove_province_modifier = ulthuan_lightly_depopulated
				remove_province_modifier = ulthuan_moderately_depopulated
				remove_province_modifier = ulthuan_heavily_depopulated
				remove_province_modifier = ulthuan_severely_depopulated
		}
	}
}

#Add the Ulthuan pop modifiers
character_event = {
	id = ulthuan.7

	hide_window = yes

	is_triggered_only = yes

	only_rulers = yes

	trigger = {
		NOT = { has_game_rule = { name = warhammer_ulthuan_depopulation value = off } }
		is_ruler = yes
		has_landed_title = e_ulthuan
		NOT = { has_global_flag = ulthuan_pop_set }
	}

	immediate = {
		ROOT = {
			any_realm_province = {
				limit = {
					region = world_ulthuan
						NOT = { has_province_modifier = ulthuan }
				}
				province_event = { id = ulthuan.0 }
			}
		}
		set_global_flag = ulthuan_pop_set
	}
}
