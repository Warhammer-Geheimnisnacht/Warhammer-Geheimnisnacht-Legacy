namespace = wh_wonders

character_event = {
	id = wh_wonders.1
	hide_window = yes

	is_triggered_only = yes

	trigger = {
		has_minor_title = title_organist_of_praag
		NOR = {
			trait = lunatic_3
		 }
	}

	immediate = {
		if = {
			limit = {
				NOT = { has_trait = lustful }
			}
			add_trait = lustful
		}
		if = {
			limit = {
				NOR = {
					has_trait = lunatic_1
					has_trait = lunatic_2
					has_trait = lunatic_3
				}
			}
			add_trait = lunatic_1
		}
		else_if = {
			limit = {
				NOR = {
					has_trait = lunatic_2
					has_trait = lunatic_3
				}
			}
			add_trait = lunatic_2
		}
		else_if = {
			limit = {
				NOT = { has_trait = lunatic_3 }
			}
			add_trait = lunatic_3
		}
	}
}
