#A Mage wanders to court
character_event = {
	id = sorcererapprentice.1
	hide_window = yes

	mean_time_to_happen = { years = 50}

	only_rulers = yes
	capable_only = yes
	min_age = 12
	max_age = 90

	trigger = { prisoner = no
			higher_tier_than = COUNT
			trait = creature_human
			NOT = { any_courtier = { OR = { has_character_modifier = magic_power_1
							has_character_modifier = magic_power_2
							has_character_modifier = magic_power_3
							has_character_modifier = magic_power_4
							 } } }
			NOT = { OR = { has_character_modifier = magic_power_1
					has_character_modifier = magic_power_2
					has_character_modifier = magic_power_3
					has_character_modifier = magic_power_4
					 } }
			NOT = { religion = taalite }
			NOT = { religion = ulrican }
			NOT = { religion = manannite }
			NOT = { religion = shallyan }
			NOT = { religion = myrmidian }
			NOT = { religion = grail }
			NOT = { religion = sigmarite }
			OR = { religion = kislev_gods
				religion_group = chaos_gods_group
				religion_group = old_gods_group
				religion_group = ormazd_gods_group }
			OR = { independent = yes
				liege = {  NOT = { religion = taalite }
						NOT = { religion = ulrican }
						NOT = { religion = manannite }
						NOT = { religion = shallyan }
						NOT = { religion = myrmidian }
						NOT = { religion = grail }
						NOT = { religion = sigmarite } } }
			}

	immediate = {
			if = { limit = { religion_group = chaos_gods_group }
				character_event = { id = sorcererapprentice.10001 }
				 }
			if = { limit = { religion_group = old_gods_group }
				character_event = { id = sorcererapprentice.10002 }
				 }
			if = { limit = { religion_group = ormazd_gods_group }
				character_event = { id = sorcererapprentice.10003 }
				 }
			if = { limit = { religion = kislev_gods }
				character_event = { id = sorcererapprentice.10004 }
				 }
	}
}

#Child wants to become a apprentice.
character_event = {
	id = sorcererapprentice.2
	desc = EVTDESCsorcererapprentice2
	picture = "GFX_evt_apprentice"

	min_age = 6
	max_age = 16

	trigger = {
		NOT = { has_character_flag = failed_mage }
		trait = creature_human
		has_epidemic = no
		is_ill = no
		gender_is_allowed_magic = yes
		OR = {
			AND = {
				guardian = { 
					is_trained_mage = yes				
				}		
			}
			AND = {	
				is_landed = yes
				any_courtier_or_vassal = {
					has_minor_title = title_teacher_mage
				}
			}
			AND = {	
				is_landed = no
				liege = {
					any_courtier_or_vassal = {
						has_minor_title = title_teacher_mage
					}
				}
			}
		}	
		NOT = { age = 15 }
		NOT = { trait = hedge_wizard }
		NOT = { trait = mage_apprentice }
		NOT = { trait = incapable }
		NOT = { has_character_modifier = apprenticeship_refused }
		is_trained_mage = no
		OR = { 
			AND = { 
				learning = 2
				age = 6 
			}
			AND = { 
				learning = 4
				age = 11 
			}
		}
		NOT = { has_character_modifier = apprenticeship_refused }
	}


	mean_time_to_happen =  { 
		years = 2

		modifier = {
			factor = 0.25
			trait = genius
		}
		modifier = {
			factor = 0.5
			trait = brilliant
		}
		modifier = {
			factor = 0.75
			trait = bright
		}
		modifier = {
			factor = 0.75
			trait = quick
		}
		modifier = {
			factor = 0.75
			trait = smart
		}
		modifier = {
			factor = 0.25
			is_ruler = yes
			any_courtier_or_vassal = {
				has_minor_title = title_teacher_mage
			}
		}
		modifier = {
			factor = 0.25
			is_ruler = no
			liege = {
				any_courtier_or_vassal = {
					has_minor_title = title_teacher_mage
				}
			}
		}
		modifier = {
			factor = 10
			OR = {
				trait = content
				trait = imbecile
				}
			}
		modifier = {
			factor = 5
			OR = { trait = slothful
				trait = idiot }
			}

	}

	option = {
		name = EVTOPTAsorcererapprentice2
		hidden_tooltip = {
			if = {
				limit = { 
					has_guardian = yes
					guardian = {  
						is_trained_mage = yes
					}
				}
				guardian = { character_event = { id = sorcererapprentice.3 } }
			}
			else_if = {
				is_landed = yes
				random_courtier_or_vassal = {
					limit = {
						has_minor_title = title_teacher_mage
						is_trained_mage = yes
					}
					character_event = { id = sorcererapprentice.3 }
				}
			}
			else_if = {
				is_landed = no
				liege = {
					random_courtier_or_vassal = {
						limit = {
							has_minor_title = title_teacher_mage
							is_trained_mage = yes
						}
						character_event = { id = sorcererapprentice.3 }
					}
				}
			}
		}
	}
}
