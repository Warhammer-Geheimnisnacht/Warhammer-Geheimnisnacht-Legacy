					break = yes }
				if = { limit = { NOT = { trait = mutation_massive }
						check_variable = { which = "mutation_counter" value = 20 } }
					remove_trait = mutation_minor
					remove_trait = mutation_major
					add_trait = mutation_massive
					character_event = { id = mutation.1002 }
					break = yes }
				if = { limit = { NOT = { trait = mutation_major }
						check_variable = { which = "mutation_counter" value = 10 } }
					remove_trait = mutation_minor
