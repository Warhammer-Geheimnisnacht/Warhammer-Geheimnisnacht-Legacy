check_if_mage_should_be_burned = {
	NOT = {  top_liege = { has_character_modifier = expelled_k_witch_hunters_order } }
	#Peasant leaders need all the help they can get and are going to have bigger concerns with organising their realms.
	NOT = { top_liege = {  trait = peasant_leader } }
	custom_tooltip = {
		text = magic_is_allowed_in_realm_tt
		OR = {
			AND = {
				has_magic_law_0 = yes
				OR = {
					has_magic_potential = yes
					is_trained_mage = yes
				}
				#The Ice witchs are tolerated in the Empire to avoid pissing off a useful ally
				NOT = { trait = ice_witch }
				NOT = { trait = BretonDamsel }
				gender_is_allowed_magic = no
			}
			AND = {
				is_trained_mage = yes
				gender_is_allowed_magic = no
				#The Ice witchs are tolerated in the Empire to avoid pissing off a useful ally
				NOT = { trait = ice_witch }
				NOT = { trait = BretonDamsel }
				OR = {
					has_magic_law_0 = yes
					has_magic_law_1 = yes
				}
			}
			AND = {
				is_trained_mage = yes
				has_magic_law_any = yes
				old_world_forbidden_lore = yes
			}
		}
		NOT = { trait = creature_elf }
	}
}

gender_is_allowed_magic = {
	custom_tooltip = {
		text = gender_allowed_magic_tt
		OR = {
			AND = {
				is_female = yes
				OR = {
					religion = kislev_gods
					religion = grail
					top_liege = { trait = malekith }
				}
			}
			AND = {
				is_female = no
				OR = {
					religion = ormazd
					religion = kitabid
				}
			}
			has_magic_law_any = no
			has_magic_law_2 = yes
		}
	}
}

supports_legal_magic = {
	OR = {
		hidden_tooltip = {
			is_human = no
		}
		is_trained_mage_or_has_magic_potential = yes
		#Made too easy to pass laws at start
		#hidden_tooltip = {
		#	secret_religion_group = chaos_gods_group
		#}
		AND = {
			#Realms with law 0 are more inclined to hate magic.
			has_magic_law_0 = yes
			hates_legal_magic = no
		}
		calc_true_if = {
			amount = 2
			custom_tooltip = {
				text = bloodline_effect_magic_legacy
				any_owned_bloodline = {
					has_bloodline_flag = magic_legacy
					bloodline_is_active_for = PREV
				}
			}
			trait = kind
			trait = just
			trait = scholar
			trait = mystic
			is_doom_high = yes
			is_doom_very_high = yes
			any_spouse = {
				is_trained_mage_or_has_magic_potential = yes
			}
			father_even_if_dead = {
				is_trained_mage_or_has_magic_potential = yes
			}
			mother_even_if_dead = {
				is_trained_mage_or_has_magic_potential = yes
			}
			any_child  = {
				is_trained_mage_or_has_magic_potential = yes
			}
			any_sibling  = {
				is_trained_mage_or_has_magic_potential = yes
			}
			any_friend = {
				OR = {
					is_trained_mage_or_has_magic_potential = yes
					trait = creature_elf
				}
			}
			any_lover = {
				OR = {
					is_trained_mage_or_has_magic_potential = yes
					trait = creature_elf
				}
			}
			has_global_flag = magic_legal_old_world_50_years
		}
	}
}

hates_legal_magic = {
	NOT = { is_trained_mage_or_has_magic_potential = yes }
	hidden_tooltip = {
		is_human = yes
	}
	OR = {
		trait = zealous
		OR = {
			AND = {
				has_magic_law_0 = yes
				NOT = { learning = 16 }
			}
			AND = {
				has_magic_law_1_or_above = yes
				NOT = { learning = 12 }
			}
		}
	}
	OR = {
		AND = {
			has_magic_law_0 = yes
			NOT = {
				has_global_flag = magic_legal_old_world_50_years
			}
			OR = {
				trait = wroth
				trait = cruel
				trait = impaler
				trait = arbitrary
				trait = paranoid
				trait = detached_priest
				trait = martial_cleric
				trait = trusting
				trait = shy
				trait = craven
				trait = envious
				ai_rationality < 0
				is_doom_very_low = yes
				is_doom_low = yes
				AND = {
					is_theocracy = yes
					NOT = { trait = kind }
				}
			}
		}
		AND = {
			OR = {
				has_magic_law_1 = yes
				AND = {
					has_magic_law_0 = yes
					has_global_flag = magic_legal_old_world_50_years
				}
			}
			OR = {
				trait = wroth
				trait = cruel
				trait = impaler
				trait = arbitrary
				trait = paranoid
				trait = detached_priest
				trait = martial_cleric
				ai_rationality < 0
				is_doom_very_low = yes
				is_doom_low = yes
				AND = {
					is_theocracy = yes
					NOT = { trait = kind }
				}
			}
		}
		AND = {
			has_magic_law_2 = yes
			calc_true_if = {
				amount = 2
				trait = wroth
				trait = cruel
				trait = impaler
				trait = arbitrary
				trait = paranoid
				trait = detached_priest
				trait = martial_cleric
				is_dumb_trigger = yes
				ai_rationality < 0
				AND = {
					is_theocracy = yes
					is_dumb_trigger = yes
				}
			}
		}
	}
}

can_grant_title_teacher_mage = {
	is_theocracy = no
	is_landed = yes
	is_playable = yes
	NOT = { religion = grail }
	OR = {
		has_magic_law_2 = yes
		religion = kislev_gods
		religion = ormazd
		religion = kitabid
		AND = {
			independent = yes
			is_trained_mage = yes
		}
	}
	OR = {
		trait = creature_human
		AND = {
			trait = creature_elf
			culture_group = asur_group
			OR = {
				NOT = {
					top_liege = {
						trait = malekith
					}
					top_liege = {
						has_landed_title = e_ulthuan
					}
				}
				AND = {
					independent = yes
					NOR = {
						has_landed_title = e_ulthuan
						trait = malekith
					}
				}
			}
		}
		trait = creature_amazon
		is_daemon_prince_trigger = yes
	}
	NOT = { religion = khorne }
}

can_hold_title_teacher_mage = {
	is_adult = yes
	is_ruler = no
	prisoner = no
	OR = {
		liege = {
			NOT = {
				religion = kislev_gods
			}
		}
		AND = {
			liege = {
				religion = kislev_gods
			}
			trait = ice_witch
		}
	}
	NOR = {
		trait = incapable
		trait = imbecile
	}
	OR = {
		trait = creature_human
		trait = creature_elf
		trait = creature_amazon
		is_daemon_prince_trigger = yes
	}
	OR = {
		has_magic_law_any = no
		AND = {
			#Ice Witches, Araby men etc
			has_magic_law_any = yes
			check_if_mage_should_be_burned = no
		}
	}
	gender_is_allowed_magic = yes
	is_trained_mage = yes
}
