can_use_retinue_trigger = {
	OR = {
		ai = no
		is_patrician = yes
		is_nomadic = yes
		AND = {
			higher_real_tier_than = BARON
			OR = {
				AND = {
					OR = {
						is_feudal = yes
						is_theocracy = yes
						is_republic = yes
					}
					# > doesn't work with demesne_size
					OR = {
						demesne_size = 2
						higher_real_tier_than = COUNT
					}
				}
				is_tribal = yes
			}
		}
	}
}
